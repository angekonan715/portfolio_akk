R Code for Stock Portfolio Analysis

# Stock Portfolio Analysis library(tidyverse) library(quantmod) library(ggplot2) # Get stock data getSymbols(c("AAPL", "GOOGL", "MSFT", "AMZN"), from="2020-01-01") # Calculate returns aapl_returns <- dailyReturn(AAPL) googl_returns <- dailyReturn(GOOGL) msft_returns <- dailyReturn(MSFT) amzn_returns <- dailyReturn(AMZN) # Portfolio analysis portfolio_data <- data.frame( Date = index(aapl_returns), AAPL = as.numeric(aapl_returns), GOOGL = as.numeric(googl_returns), MSFT = as.numeric(msft_returns), AMZN = as.numeric(amzn_returns) ) # Calculate correlation matrix correlation_matrix <- cor(portfolio_data[,-1], use="complete.obs") # Portfolio performance metrics portfolio_summary <- portfolio_data %>% gather(key = "Stock", value = "Return", -Date) %>% group_by(Stock) %>% summarise( Mean_Return = mean(Return, na.rm = TRUE), Volatility = sd(Return, na.rm = TRUE), Sharpe_Ratio = Mean_Return / Volatility )
# Python equivalent for Stock Portfolio Analysis import yfinance as yf import pandas as pd import numpy as np import matplotlib.pyplot as plt import seaborn as sns # Get stock data tickers = ['AAPL', 'GOOGL', 'MSFT', 'AMZN'] data = yf.download(tickers, start='2020-01-01')['Adj Close'] # Calculate returns returns = data.pct_change() # Portfolio analysis portfolio_data = returns.copy() # Calculate correlation matrix correlation_matrix = portfolio_data.corr() # Portfolio performance metrics portfolio_summary = portfolio_data.agg(['mean', 'std']).T portfolio_summary['Sharpe_Ratio'] = portfolio_summary['mean'] / portfolio_summary['std']

# GDP per Capita Analysis library(tidyverse) library(gapminder) # Load and prepare data gdp_data <- gapminder %>% filter(year >= 2000) %>% select(country, year, gdpPercap, continent) # Analysis by continent continent_summary <- gdp_data %>% group_by(continent, year) %>% summarise( mean_gdp = mean(gdpPercap, na.rm = TRUE), median_gdp = median(gdpPercap, na.rm = TRUE), sd_gdp = sd(gdpPercap, na.rm = TRUE) ) # Top countries by GDP per capita top_countries <- gdp_data %>% filter(year == max(year)) %>% arrange(desc(gdpPercap)) %>% head(20)
# Python equivalent for GDP per Capita Analysis import pandas as pd import numpy as np import matplotlib.pyplot as plt import seaborn as sns # Load data (assuming you have gapminder data) # gdp_data = pd.read_csv('gapminder.csv') # Filter and prepare data gdp_data = gdp_data[gdp_data['year'] >= 2000][['country', 'year', 'gdpPercap', 'continent']] # Analysis by continent continent_summary = gdp_data.groupby(['continent', 'year'])['gdpPercap'].agg(['mean', 'median', 'std']).reset_index() # Top countries by GDP per capita latest_year = gdp_data['year'].max() top_countries = gdp_data[gdp_data['year'] == latest_year].nlargest(20, 'gdpPercap')
