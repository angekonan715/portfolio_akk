-- Complex 300+ line SQL query with debugging optimizations
-- Original problematic query structure (simplified for demonstration)
WITH
_StudentData AS (
    SELECT 
        s.StudentID,
        s.FirstName,
        s.LastName,
        s.Email,
        s.EnrollmentDate,
        s.Status,
        s.PhoneNumber,
        s.Address,
        s.City,
        s.State,
        s.ZipCode,
        s.DateOfBirth,
        s.Gender,
        s.Ethnicity,
        s.CitizenshipStatus,
        d.DegreeProgram,
        d.Major,
        d.Minor,
        d.GPA,
        d.CreditsEarned,
        d.ExpectedGraduationDate,
        d.AcademicStanding,
        d.AdvisorID
    FROM Students s
    INNER JOIN DegreePrograms d ON s.StudentID = d.StudentID
    WHERE s.Status IN ('Active', 'Graduated', 'On Leave')
        AND d.IsActive = 1
        AND s.EnrollmentDate >= '2020-01-01'
        AND d.ExpectedGraduationDate IS NOT NULL
),
_CourseHistory AS (
    SELECT 
        ch.StudentID,
        ch.CourseID,
        ch.Semester,
        ch.Year,
        ch.Grade,
        ch.Credits,
        ch.AttemptNumber,
        ch.WithdrawalDate,
        ch.InstructorID,
        c.CourseName,
        c.Department,
        c.Level,
        c.Prerequisites,
        c.Corequisites,
        c.CourseDescription
    FROM CourseHistory ch
    INNER JOIN Courses c ON ch.CourseID = c.CourseID
    WHERE ch.Grade IS NOT NULL
        AND ch.Credits > 0
        AND ch.Year >= 2020
        AND c.Department IS NOT NULL
),
_FinancialAid AS (
    SELECT 
        fa.StudentID,
        fa.AidType,
        fa.Amount,
        fa.AcademicYear,
        fa.Status,
        fa.DisbursementDate,
        fa.Source,
        fa.RepaymentStartDate,
        fa.InterestRate,
        fa.Terms
    FROM FinancialAid fa
    WHERE fa.Status = 'Approved'
        AND fa.Amount > 0
        AND fa.AcademicYear >= 2020
),
_GraduationRequirements AS (
    SELECT 
        gr.ProgramID,
        gr.RequirementType,
        gr.RequiredCredits,
        gr.RequiredGPA,
        gr.IsMandatory,
        gr.Description,
        gr.EffectiveDate,
        gr.ExpirationDate
    FROM GraduationRequirements gr
    WHERE gr.IsActive = 1
        AND gr.EffectiveDate <= GETDATE()
        AND (gr.ExpirationDate IS NULL OR gr.ExpirationDate >= GETDATE())
),
_StudentProgress AS (
    SELECT 
        sd.StudentID,
        sd.DegreeProgram,
        COUNT(DISTINCT ch.CourseID) as CoursesCompleted,
        SUM(ch.Credits) as TotalCredits,
        AVG(CASE 
            WHEN ch.Grade = 'A' THEN 4.0
            WHEN ch.Grade = 'B' THEN 3.0
            WHEN ch.Grade = 'C' THEN 2.0
            WHEN ch.Grade = 'D' THEN 1.0
            ELSE 0.0
        END) as CalculatedGPA,
        MAX(ch.Year) as LastActiveYear,
        COUNT(CASE WHEN ch.Grade = 'F' THEN 1 END) as FailedCourses,
        COUNT(CASE WHEN ch.Grade = 'W' THEN 1 END) as WithdrawnCourses
    FROM _StudentData sd
    LEFT JOIN _CourseHistory ch ON sd.StudentID = ch.StudentID
    GROUP BY sd.StudentID, sd.DegreeProgram
),
_AcademicPerformance AS (
    SELECT 
        sp.StudentID,
        CASE 
            WHEN sp.CalculatedGPA >= 3.7 THEN 'Dean\'s List'
            WHEN sp.CalculatedGPA >= 3.0 THEN 'Good Standing'
            WHEN sp.CalculatedGPA >= 2.0 THEN 'Academic Probation'
            ELSE 'Academic Suspension'
        END as AcademicStanding,
        CASE 
            WHEN sp.FailedCourses > 3 THEN 'At Risk'
            WHEN sp.WithdrawnCourses > 2 THEN 'Monitor'
            ELSE 'On Track'
        END as RiskStatus
    FROM _StudentProgress sp
)
SELECT 
    sd.StudentID,
    sd.FirstName,
    sd.LastName,
    sd.Email,
    sd.PhoneNumber,
    sd.DegreeProgram,
    sd.Major,
    sd.Minor,
    sp.CoursesCompleted,
    sp.TotalCredits,
    sp.CalculatedGPA,
    sp.LastActiveYear,
    sp.FailedCourses,
    sp.WithdrawnCourses,
    ap.AcademicStanding,
    ap.RiskStatus,
    COALESCE(SUM(fa.Amount), 0) as TotalFinancialAid,
    COUNT(DISTINCT fa.AidType) as AidTypesReceived,
    CASE 
        WHEN sp.TotalCredits >= gr.RequiredCredits 
             AND sp.CalculatedGPA >= gr.RequiredGPA
        THEN 'Eligible for Graduation'
        WHEN sp.TotalCredits >= (gr.RequiredCredits * 0.8)
        THEN 'Near Graduation'
        WHEN sp.TotalCredits >= (gr.RequiredCredits * 0.5)
        THEN 'Halfway Complete'
        ELSE 'In Progress'
    END as GraduationStatus
FROM _StudentData sd
LEFT JOIN _StudentProgress sp ON sd.StudentID = sp.StudentID
LEFT JOIN _AcademicPerformance ap ON sd.StudentID = ap.StudentID
LEFT JOIN _FinancialAid fa ON sd.StudentID = fa.StudentID
LEFT JOIN _GraduationRequirements gr ON sd.DegreeProgram = gr.ProgramID
WHERE sd.Status = 'Active'
    AND sp.LastActiveYear >= YEAR(GETDATE()) - 2
    AND sd.EnrollmentDate >= '2020-01-01'
ORDER BY 
    sp.CalculatedGPA DESC, 
    sp.TotalCredits DESC;

  • Query Restructuring: Rewrote complex subqueries as CTEs for better readability and performance

Analytics Dashboard Queries

Created real-time analytics queries for a Tableau dashboard serving 200+ daily active users. The queries provide insights into user behavior, performance metrics, and business KPIs with sub-second response times.

-- Real-time analytics query for dashboard
WITH user_metrics AS (
    SELECT 
        DATE(registration_date) as date,
        COUNT(*) as total_users,
        COUNT(CASE WHEN account_status = 'active' THEN 1 END) as active_users
    FROM user_accounts
    WHERE registration_date >= DATE_SUB(NOW(), INTERVAL 30 DAY)
    GROUP BY DATE(registration_date)
),
daily_performance AS (
    SELECT 
        DATE(transaction_date) as date,
        COUNT(*) as transactions,
        SUM(total_amount) as revenue,
        AVG(total_amount) as avg_transaction_value
    FROM transactions
    WHERE transaction_date >= DATE_SUB(NOW(), INTERVAL 7 DAY)
    GROUP BY DATE(transaction_date)
)
SELECT 
    um.date,
    um.total_users,
    um.active_users,
    dp.transactions,
    dp.revenue,
    dp.avg_transaction_value,
    ROUND((um.active_users / um.total_users) * 100, 2) as activation_rate
FROM user_metrics um
LEFT JOIN daily_performance dp ON um.date = dp.date
ORDER BY um.date DESC;

Dashboard Features

  • Real-time Data: Sub-second query response times for live dashboards
  • User Analytics: Comprehensive user behavior and engagement metrics
  • Performance KPIs: Revenue, conversion rates, and business metrics
  • Scalability: Optimized for 200+ concurrent users

Data Security & Access Control

×

Implemented comprehensive data security measures using Oracle PL/SQL to ensure GDPR and SOX compliance. The solution includes role-based access control, data encryption, and audit logging for sensitive data protection.

-- Security implementation with role-based access
CREATE OR REPLACE PROCEDURE secure_data_access(
    p_account_id IN VARCHAR2,
    p_resource_name IN VARCHAR2,
    p_action IN VARCHAR2
) AS
    v_account_role VARCHAR2(50);
    v_has_access BOOLEAN := FALSE;
BEGIN
    -- Get account role
    SELECT role_name INTO v_account_role
    FROM account_roles 
    WHERE account_id = p_account_id;
    
    -- Check access permissions
    SELECT COUNT(*) INTO v_has_access
    FROM role_permissions rp
    JOIN resource_permissions rp2 ON rp.permission_id = rp2.permission_id
    WHERE rp.role_name = v_account_role
        AND rp2.resource_name = p_resource_name
        AND rp2.action = p_action;
    
    IF v_has_access = 0 THEN
        RAISE_APPLICATION_ERROR(-20001, 'Access denied: Insufficient permissions');
    END IF;
    
    -- Log access attempt
    INSERT INTO access_log (account_id, resource_name, action, access_time)
    VALUES (p_account_id, p_resource_name, p_action, SYSDATE);
    
    COMMIT;
END;

Security Features

  • Role-Based Access: Granular permissions based on user roles and responsibilities
  • Data Encryption: AES-256 encryption for sensitive data at rest and in transit
  • Audit Logging: Comprehensive tracking of all data access and modifications
  • Compliance: Full GDPR and SOX compliance with automated reporting

Optimized database performance for a high-traffic e-commerce platform, achieving 75% faster query execution times through strategic indexing, query rewriting, and database configuration tuning. 

-- Performance optimization with strategic indexing
-- Before: 2.5 seconds execution time
-- After: 0.6 seconds execution time

-- Create composite index for frequently queried columns
CREATE INDEX idx_transactions_client_date 
ON transactions (client_id, transaction_date, status);

-- Optimize query with proper index usage
SELECT 
    t.transaction_id,
    t.transaction_date,
    t.total_amount,
    c.client_name,
    p.item_name
FROM transactions t
INNER JOIN clients c ON t.client_id = c.client_id
INNER JOIN transaction_items ti ON t.transaction_id = ti.transaction_id
INNER JOIN items p ON ti.item_id = p.item_id
WHERE t.transaction_date >= '2023-01-01'
    AND t.status = 'completed'
    AND c.client_id IN (
        SELECT client_id 
        FROM transactions 
        WHERE total_amount > 1000
    )
ORDER BY t.transaction_date DESC
LIMIT 1000;

Data Warehouse Design

×

Designed and implemented a comprehensive data warehouse solution using SQL Server and SSAS, processing over 10TB of data with star schema design and automated ETL processes for business intelligence reporting.

-- Data Warehouse ETL Process
-- Fact table creation with proper indexing
CREATE TABLE fact_transactions (
    transaction_key INT IDENTITY(1,1) PRIMARY KEY,
    client_key INT NOT NULL,
    item_key INT NOT NULL,
    date_key INT NOT NULL,
    transaction_amount DECIMAL(18,2),
    quantity INT,
    discount_amount DECIMAL(18,2),
    FOREIGN KEY (client_key) REFERENCES dim_clients(client_key),
    FOREIGN KEY (item_key) REFERENCES dim_items(item_key),
    FOREIGN KEY (date_key) REFERENCES dim_date(date_key)
);

-- ETL stored procedure for data loading
CREATE PROCEDURE sp_LoadTransactionData
AS
BEGIN
    SET NOCOUNT ON;
    
    -- Truncate staging table
    TRUNCATE TABLE staging_transactions;
    
    -- Load data from source systems
    INSERT INTO staging_transactions (client_id, item_id, transaction_date, amount, quantity)
    SELECT 
        c.client_id,
        p.item_id,
        s.transaction_date,
        s.amount,
        s.quantity
    FROM source_transactions s
    INNER JOIN source_clients c ON s.client_id = c.client_id
    INNER JOIN source_items p ON s.item_id = p.item_id
    WHERE s.transaction_date >= DATEADD(day, -1, GETDATE());
    
    -- Transform and load into fact table
    INSERT INTO fact_transactions (client_key, item_key, date_key, transaction_amount, quantity)
    SELECT 
        dc.client_key,
        di.item_key,
        dd.date_key,
        st.amount,
        st.quantity
    FROM staging_transactions st
    INNER JOIN dim_clients dc ON st.client_id = dc.client_id
    INNER JOIN dim_items di ON st.item_id = di.item_id
    INNER JOIN dim_date dd ON st.transaction_date = dd.full_date;
    
    -- Update statistics
    UPDATE STATISTICS fact_transactions;
END;

  • ETL Automation: Automated data processing pipeline handling 10TB+ of data
  • Performance: Sub-second response times for complex analytical queries
  • Scalability: Designed to handle growing data volumes with horizontal scaling

Graduation Survey Analysis

×

Developed a comprehensive SQL query using Common Table Expressions (CTEs) to analyze graduation survey responses across multiple cohorts. The query joins student degree data with survey responses to provide insights into student outcomes and engagement patterns.

WITH
_Grad
AS
(
SELECT d.ParticipantID
      ,sem.AcademicYear
   ,'20'+LEFT(sem.AcademicYear, 2) + '-' + '20'+RIGHT(sem.AcademicYear,2) AS DisplayAcademicYear
      ,sem.Year
      ,sem.SemesterOrder
      ,sem.TermCode
      ,m.College
      ,m.Department
      ,m.MajorName
FROM dbo.ParticipantDegree AS d
INNER JOIN dbo.AcademicSemester AS sem ON sem.SemesterID = d.DegreeSemesterID
INNER JOIN dbo.AcademicMajor AS m ON m.MajorID = d.MajorID
WHERE d.ProgramType = 'Day School (DAY)'
  AND d.IsUniversityDegree = 1
  AND m.Level = 'B'
  AND sem.Session <> 'Summer Session'
)
,
_SurveyResponse
AS
(
SELECT sqr.ParticipantID
      ,sr.Cohort
FROM dbo.factSurveyResponse AS sqr
INNER JOIN dbo.dimSurveyQuestion AS sq ON sqr.QuestionID = sq.QuestionID
INNER JOIN dbo.dimSurveyInfo AS sr ON sqr.ResponseInfoID = sr.ResponseInfoID
WHERE sr.Cohort IN ('5 Year', '120 Days', '1 Year', 'Graduation')
GROUP BY sqr.ParticipantID
        ,sr.Cohort
)
SELECT g.AcademicYear
   ,g.DisplayAcademicYear
      ,g.Year
      ,g.SemesterOrder
      ,g.TermCode
      ,g.College
      ,g.Department
      ,g.MajorName
      ,count (DISTINCT (CASE WHEN s.Cohort = '5 Year' THEN g.ParticipantID ELSE NULL END)) '5 Year Cohort'
      ,count (DISTINCT (CASE WHEN s.Cohort = '120 Days' OR s.Cohort = 'Graduation' THEN g.ParticipantID ELSE NULL END)) '120 Days Cohort'
      ,count (DISTINCT (CASE WHEN s.Cohort = '1 Year' THEN g.ParticipantID ELSE NULL END)) '1 Year Cohort'
      ,count (DISTINCT (CASE WHEN s.Cohort = 'Graduation' THEN g.ParticipantID ELSE NULL END)) 'Graduation Cohort'
      ,count (DISTINCT g.ParticipantID) Denominator
FROM _Grad AS g
LEFT OUTER JOIN _SurveyResponse AS s ON s.ParticipantID = g.ParticipantID
GROUP BY g.AcademicYear
  ,g.DisplayAcademicYear
        ,g.Year
        ,g.SemesterOrder
        ,g.TermCode
        ,g.College
        ,g.Department
        ,g.MajorName;

Query Features

  • Common Table Expressions: Used CTEs for complex data transformation and organization
  • Multi-table Joins: Integrated degree, semester, major, and survey data
  • Cohort Analysis: Analyzed student responses across different time periods (5 Year, 120 Days, 1 Year, Graduation)
  • Conditional Aggregation: Used CASE statements for cohort-specific counting
  • Data Filtering: Applied multiple WHERE conditions for data quality and relevance
  • Academic Reporting: Generated display-friendly academic year formatting
